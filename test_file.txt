    <script>
        // Use the polling logic from before...
        function refreshDashboardStats() {
            fetch('/dashboard-stats/') 
                .then(response => response.json())
                .then(data => {
                    updateBadge('badge-new-emails', data.new_emails_count);
                    updateBadge('badge-recycled', data.recycled_count);
                    updateBadge('badge-manager-approval', data.pending_approval_count);
                    updateBadge('badge-my-delegations', data.my_pending_delegations_count);
                    updateBadge('badge-bank-lines', data.bank_lines_count);
                    updateBadge('badge-credit-notes', data.credit_notes_count);
                });
        }

        function updateBadge(id, count) {
            const badge = document.getElementById(id);
            if (!badge) return;
            badge.innerText = count;
            badge.style.display = count > 0 ? 'flex' : 'none';
        }
        setInterval(refreshDashboardStats, 30000);
    </script>
